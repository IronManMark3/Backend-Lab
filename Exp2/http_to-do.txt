# Create
curl -X POST http://localhost:3000/items -H "Content-Type: application/json" -d '{"title":"Buy milk","done":false}'

# List
curl http://localhost:3000/items

# Read one
curl http://localhost:3000/items/1

# Update (PUT = full replace)
curl -X PUT http://localhost:3000/items/1 -H "Content-Type: application/json" -d '{"title":"Buy milk & bread","done":true}'

# Delete
curl -X DELETE http://localhost:3000/items/1

# Verify empty
curl http://localhost:3000/items
